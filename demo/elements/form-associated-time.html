Form Associated Time Element

A custom element with an interface to manipulate time values with form association.

Example

example.html

<a-form-associated-time name="time" value="15:15"></a-form-associated-time>

example.js

import { FormAssociatedTimeElement } from "https://cdn.jsdelivr.net/npm/@jsxtools/aui/dist/elements/form-associated-time.js"

customElements.define("a-form-associated-time", FormAssociatedTimeElement)

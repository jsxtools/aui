Toggle Group

Provides a shared state to a series of toggle buttons.

<a-toggle-group>
	<a-toggle>
		<svg><use href="#:icon:text-align-left" /></svg>
	</a-toggle>
	<a-toggle>
		<svg><use href="#:icon:text-align-center" /></svg>
	</a-toggle>
	<a-toggle>
		<svg><use href="#:icon:text-align-right" /></svg>
	</a-toggle>
</a-toggle-group>

<script>
import { ToggleGroupElement } from "@jsxtools/aui/elements/toggle-group"
import { ToggleElement } from "@jsxtools/aui/elements/toggle"

customElements.define("a-toggle-group", ToggleGroupElement)
customElements.define("a-toggle", ToggleElement)
</script>

Accordion Element

A custom element that provides collapsible sections by headings.

Ingredients

A list of ingredients.

Instructions

A list of instructions.

Additional information

A list of additional information.

example.html

<a-accordion>
	<h3>Ingredients</h3>
	<p>A list of ingredients.</p>

	<h3>Instructions</h3>
	<p>A list of instructions.</p>

	<h3>Additional information</h3>
	<p>A list of additional information.</p>
</a-accordion>

example.js

import { AccordionElement } from "https://cdn.jsdelivr.net/npm/@jsxtools/aui/dist/elements/accordion.js"

customElements.define("a-accordion", AccordionElement)
